107:     {
108:         $url = '/o/' . rawurlencode($this->config->orgName);
109:         $this->setBaseUrl($url);
110:         $appList = array();

112:             $lastKey = null;
113:             while (true) {
114:                 $queryString = 'expand=true&rows=' . $this->pageSize;
115:                 if (isset($lastKey)) {
116:                     $queryString .= '&lastKey=' . urlencode($lastKey);
117:                 }
118:                 $this->get($queryString);
119:                 $appSubset = $this->responseObj;
120:                 if (!array_key_exists('app', $appSubset)) {
121:                     break;
122:                 }
123:                 $subsetCount = count($appSubset['app']);
124:                 if ($subsetCount == 0) {
125:                     break;
126:                 }
127:                 if (isset($lastKey)) {
128:                     // Avoid duplicating the last key, which is the first key
129:                     // on this page.
130:                     array_shift($appSubset['app']);
131:                 }
132:                 foreach ($appSubset['app'] as $appDetail) {
133:                     if (array_key_exists('developerId', $appDetail)) {
134:                         $ownerId = $this->getDeveloperMailById($appDetail['developerId']);

186:      * Loads a DeveloperApp/CompanyApp, given its appId (which is a UUID).
187:      *
188:      * Normally you'd find an app by listing its owner entity's apps and looking
189:      * for the name you want. However, if you already know the app's unique id,
190:      * you can load without knowing its owner.
191:      *
192:      * If you pass true as the second parameter here, the DeveloperApp/CompanyApp
193:      * object will be changed so that it pulls apps from this developer/company
194:      * by default.
195:      *
196:      * @param string $appId

198:      * @return \Apigee\ManagementAPI\AbstractApp

201:      */

203:     {
204:         if (!preg_match('!^[0-9a-f]{8}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{12}$!', $appId)) {
205:             throw new ParameterException('Invalid UUID “' . $appId . '” passed as appId.');
206:         }
207: 
208:         $url = '/o/' . rawurlencode($this->config->orgName) . '/apps';
209:         $this->setBaseUrl($url);
210:         $this->get($appId);
211:         $this->restoreBaseUrl();
212:         $response = $this->responseObj;
213:         if (array_key_exists('developerId', $response)) {
214:             $ownerId = $this->getDeveloperMailById($response['developerId']);

236:     /**
237:      * Attempts to fetch the email address associated with a developerId.
238:      *
239:      * If no such developer exists, null is returned. We turn off all logging,
240:      * both by the main logger and by any subscribers. (The exception is that
241:      * non-404 ResponseExceptions are logged.) It is therefore the
242:      * responsibility of any client of this method to handle appropriate
243:      * logging.
244:      *
245:      * @param string $id
246:      *   The developerId of the developer in question
247:      *
248:      * @return string|null
249:      *   The email address of the developer, or null if no such developer
250:      *   exists.
251:      */
252:     private function getDeveloperMailById($id)
253:     {
254:         static $devs = array();
255:         if (!isset($devs[$id])) {
256:             $cached_logger = self::$logger;
257:             $config = clone $this->config;
258:             // Suppress (almost) all logging.
259:             $config->logger = new NullLogger();
260:             $config->subscribers = array();
261:             $dev = new Developer($config);
262:             try {
263:                 $dev->load($id);
264:                 $devs[$id] = $dev->getEmail();
265:             } catch (ResponseException $e) {
266:                 $devs[$id] = null;
267:                 // Log exceptions that are NOT 404s.
268:                 if ($e->getCode() != 404) {
269:                     $warning = 'Attempt to load dev “%s” resulted in response code of %d.';
270:                     $warningArgs = array($id, $e->getCode());
271:                     $cached_logger->warning(vsprintf($warning, $warningArgs));
272:                 }
273:             }
274:             self::$logger = $cached_logger;
275:         }
276:         return $devs[$id];
277:     }
