331:     /**
332:      * Intercepts any snake_case method invocations that aren't already
333:      * defined, turns them into camelCase, and tries to invoke them.
334:      *
335:      * Incoming snake_case method names must contain no uppercase to
336:      * qualify for this transmogrification. In the interest of efficiency,
337:      * we also don't process any method names not containing an underscore.
338:      *
339:      * @TODO When we require PHP 5.4, make this a mix-in.
340:      *
341:      * @param string $method
342:      * @param array $args
343:      * @return mixed
344:      * @throws \Apigee\Exceptions\IllegalMethodException
345:      * @internal
346:      */
347:     public function __call($method, $args)
348:     {
349:         $class = get_class();
350: 
351:         if ($method == strtolower($method) && strpos($method, '_') !== false) {
352:             $parts = explode('_', $method);
353:             $camel_case = array_shift($parts);
354:             foreach ($parts as $part) {
355:                 $camel_case .= ucfirst($part);
356:             }
357:             if (method_exists($this, $camel_case)) {
358:                 self::warnDeprecated($class, $method);
359:                 return call_user_func_array(array($this, $camel_case), $args);
360:             }
361:             throw new IllegalMethodException('Class ' . $class . ' contains no such method ' . $method . ' (even after camelCasing)');
362:         }
363:         throw new IllegalMethodException('Class ' . $class . ' contains no such method ' . $method . '');
364:     }

496:     /**
497:      * If payload is not already a string, convert it to a string based on its content-type.
498:      *
499:      * @static
500:      * @param $content_type
501:      * @param $payload
502:      */
503:     protected static function preparePayload($content_type, &$payload)
504:     {
505:         // If content_type includes charset, strip it off.
506:         if (($i = strpos($content_type, ';')) !== false) {
507:             $content_type = trim(substr($content_type, 0, $i));
508:         }
509:         if ($content_type == 'application/json' && (is_object($payload) || is_array($payload))) {
510:             // Turn objects/arrays into JSON strings.
511:             $payload = json_encode($payload);
512:         } elseif ($content_type == 'application/xml') {
513:             // Turn XML document representations into strings.
514:             if ($payload instanceof \DOMDocument) {
515:                 $payload = $payload->saveXML($payload->documentElement);
516:             } elseif ($payload instanceof \SimpleXMLElement) {
517:                 $payload = $payload->asXML();
518:                 // strip off processing instruction if present
519:                 $payload = preg_replace('!^<\?[^?]+\?>!', '', $payload);
