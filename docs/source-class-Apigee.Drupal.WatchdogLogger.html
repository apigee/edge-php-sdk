 20:   /**
 21:    * Logs an event to watchdog
 22:    *
 23:    * @param string $level
 24:    * @param mixed $message
 25:    * @param array $context
 26:    * @return void
 27:    */
 28:   public function log($level, $message, array $context = array()) {
 29:     // Translate Psr\LogLevel constants to WATCHDOG_* constants
 30:     $severity = self::log2drupal($level);
 31:     // Short-circuit if this event is too insignificant to log.
 32:     if ($severity < self::$logThreshold) {
 33:       return;
 34:     }
 35: 
 36:     // Determine how we're going to handle this.
 37:     $use_watchdog_exception = FALSE;
 38:     if ($message instanceof \Exception && !method_exists($message, '__toString')) {
 39:       // use watchdog_exception ONLY for exceptions that don't have an explicit
 40:       // __toString() method.
 41:       $use_watchdog_exception = TRUE;
 42:     }
 43:     elseif (is_array($message) || is_object($message)) {
 44:       // massage non-strings into loggable strings.
 45:       if (is_object($message) && method_exists($message, '__toString')) {
 46:         $message = (string)$message;
 47:       }
 48:       else {
 49:         ob_start();
 50:         var_dump($message);
 51:         $message = ob_get_clean();
 52:       }
 53:     }
 54: 
 55:     // Find the "type" (source) of the log request. Generally this is a class
 56:     // or file name. It may be specified in $context, or it can be derived from
 57:     // a debug_backtrace.
 58:     if (isset($context['type'])) {
 59:       $type = $context['type'];
 60:     }
 61:     else {
 62:       if (version_compare(PHP_VERSION, '5.4', '>=')) {
 63:         // Be more efficient when running PHP 5.4
 64:         $backtrace = debug_backtrace(DEBUG_BACKTRACE_IGNORE_ARGS, 2);
 65:       }
 66:       else {
 67:         // Sigh. Pull entire backtrace.
 68:         $backtrace = debug_backtrace(DEBUG_BACKTRACE_IGNORE_ARGS);
 69:       }
 70:       $type = basename($backtrace[1]['file']);
 71:       $type = preg_replace('!\.(module|php)$!', '', $type);
 72:     }
 73: 
 74:     if ($use_watchdog_exception) {
 75:       watchdog_exception($type, $message, NULL, array(), $severity);
 76:     }
 77:     else {
 78:       watchdog($type, $message, array(), $severity);
 79:     }
 80: 
 81:   }

 85:    *
 86:    * @param string $level
 87:    * @return int
 88:    */
 89:   private static function log2drupal($level) {
 90:     switch ($level) {

 92:         $level = WATCHDOG_ALERT;
 93:         break;
 94:       case LogLevel::CRITICAL:
 95:         $level = WATCHDOG_CRITICAL;
 96:         break;
 97:       case LogLevel::DEBUG:
 98:         $level = WATCHDOG_DEBUG;
 99:         break;
100:       case LogLevel::ERROR:
101:         $level = WATCHDOG_ERROR;
102:         break;
103:       case LogLevel::EMERGENCY:
104:         $level = WATCHDOG_EMERGENCY;
105:         break;
106:       case LogLevel::INFO:
107:         $level = WATCHDOG_INFO;
108:         break;
109:       case LogLevel::NOTICE:
110:         $level = WATCHDOG_NOTICE;
111:         break;
112:       case LogLevel::WARNING:
113:         $level = WATCHDOG_WARNING;
114:         break;
115:       default:

117:     }
